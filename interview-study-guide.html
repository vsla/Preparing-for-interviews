1. JavaScript (Deep) 11 topics

JavaScript is the most-tested language in frontend interviews. This section covers the subtle runtime behaviors — coercion, closures, the event loop, prototype chains, and async patterns — that interviewers use to distinguish strong candidates. Expect at least one deep JS question in any frontend or fullstack interview.

1.1 Core Types & Coercion

Primitive Types

JavaScript has 8 types: string, number, bigint, boolean, undefined, null, symbol, and object (everything else). The first 7 are primitives — immutable and compared by value.

1.2 Scope & Closures

Lexical vs Dynamic Scope

JavaScript uses lexical (static) scope — where a function is defined determines what variables it can access, not where it's called. Dynamic scope (as in Bash) would use the call site instead.

Hoisting

// var: declaration hoisted, initialization is undefined
console.log(x); // undefined (not ReferenceError)
var x = 5;
// Equivalent to: var x; console.log(x); x = 5;

// let/const: declaration hoisted but NOT initialized → TDZ
console.log(y); // ReferenceError: Cannot access 'y' before initialization
let y = 5;
// The Temporal Dead Zone (TDZ) is from block entry to the declaration

// Function declarations: fully hoisted (both name and body)
greet(); // "hello" — works!
function greet() { console.log("hello"); }

// Function expressions: only var is hoisted (as undefined)
sayHi(); // TypeError: sayHi is not a function
var sayHi = function() { console.log("hi"); };

// Arrow functions: same as function expressions
go(); // TypeError
var go = () => {};

Closures

A closure is a function that remembers the variables from its outer scope even after the outer function has returned.

// Classic closure: counter
function makeCounter(start = 0) {
  let count = start;
  return {
    increment() { return ++count; },
    decrement() { return --count; },
    value()     { return count; }
  };
}
const c = makeCounter(10);
c.increment(); // 11
c.increment(); // 12
c.value();     // 12

// Memoization via closure
function memoize(fn) {
  const cache = new Map();
  return function(...args) {
    const key = JSON.stringify(args);
    if (cache.has(key)) return cache.get(key);
    const result = fn.apply(this, args);
    cache.set(key, result);
    return result;
  };
}

// Partial application via closure
function multiply(a, b) { return a * b; }
const double = memoize(multiply.bind(null, 2));
double(5); // 10

// Module pattern
const BankAccount = (function() {
  let balance = 0; // private
  return {
    deposit(amt)  { balance += amt; },
    withdraw(amt) { if (amt <= balance) balance -= amt; },
    getBalance()  { return balance; }
  };
})();

IIFE (Immediately Invoked Function Expression)

(function() {
  var private = "not polluting global scope";
  console.log(private);
})();

// Arrow IIFE
(() => {
  const x = 42;
})();

let in Loops — Classic Interview Trap

// var: all callbacks share the SAME variable
for (var i = 0; i < 3; i++) {
  setTimeout(() => console.log(i), 0);
}
// Output: 3 3 3  (i is 3 by the time callbacks run)

// let: each iteration gets its OWN binding

// Output: 0 1 2  ✓

// var workaround with IIFE

  (function(j) {
    setTimeout(() => console.log(j), 0);
  })(i);
}
// Output: 0 1 2
1.3 this & Context

The 4 Rules of this (in order of precedence)

1. new binding

function Person(name) { this.name = name; }
const p = new Person("Alice");
// 1. new object created
// 2. [[Prototype]] linked
// 3. this bound to new object
// 4. object returned (unless constructor returns object)

2. Explicit binding (call/apply/bind)

function greet() { return `Hello, ${this.name}`; }
const obj = { name: "Bob" };

greet.call(obj);           // "Hello, Bob"  — call(thisArg, arg1, arg2)
greet.apply(obj);          // "Hello, Bob"  — apply(thisArg, [args])
const boundGreet = greet.bind(obj);
boundGreet();              // "Hello, Bob"  — returns new function

// bind vs call vs apply
Math.max.call(null, 1, 2, 3);    // 3
Math.max.apply(null, [1, 2, 3]); // 3  — useful when args are in array

// Partial application with bind
function add(a, b) { return a + b; }
const add5 = add.bind(null, 5);
add5(3); // 8

3. Implicit binding

const obj = {
  name: "Carol",
  greet() { return `Hello, ${this.name}`; }
};
obj.greet(); // "Hello, Carol" — this = obj (call site has context object)

// Losing implicit binding
const fn = obj.greet;
fn(); // "Hello, undefined" — this = global (or undefined in strict mode)

// Fix: bind
const fn2 = obj.greet.bind(obj);
fn2(); // "Hello, Carol"

// Losing this in callbacks
setTimeout(obj.greet, 100);         // broken
setTimeout(obj.greet.bind(obj), 100); // fixed
setTimeout(() => obj.greet(), 100);   // also fixed

4. Default binding

function whoAmI() { return this; }
whoAmI();          // window (sloppy mode) or undefined (strict mode)
// "use strict" at top of file or function prevents default binding

Arrow Functions: Lexical this

class Timer {
  constructor() {
    this.seconds = 0;
  }
  start() {
    // Arrow function: inherits this from start() method's this
    setInterval(() => {
      this.seconds++;
      console.log(this.seconds);
    }, 1000);
  }
}

// vs. regular function (broken):
class BrokenTimer {
  start() {
    setInterval(function() {
      this.seconds++; // this is undefined in strict mode!
    }, 1000);
  }
}

// Arrow functions CANNOT be used as constructors
const Foo = () => {};
new Foo(); // TypeError: Foo is not a constructor

// Arrow functions have no 'arguments' object
const f = () => { console.log(arguments); }; // ReferenceError

this in Classes

class Animal {
  constructor(name) {
    this.name = name;
  }
  speak() {
    return `${this.name} makes a noise.`;
  }
}
class Dog extends Animal {
  speak() {
    return `${this.name} barks.`;
  }
}
const d = new Dog("Rex");
d.speak(); // "Rex barks."

// Class methods lose this when passed as callbacks
const { speak } = d;

1.4 Prototypes & Inheritance

The Prototype Chain

// Every object has [[Prototype]] (exposed as __proto__, non-standard)
const arr = [1, 2, 3];
// arr → Array.prototype → Object.prototype → null

// Functions have a .prototype property used for inheritance
function Animal(name) { this.name = name; }
Animal.prototype.speak = function() { return this.name; };

const dog = new Animal("Rex");
dog.speak();                        // "Rex" — found on Animal.prototype
dog.hasOwnProperty("name");         // true
dog.hasOwnProperty("speak");        // false — inherited
Object.getPrototypeOf(dog) === Animal.prototype; // true
dog instanceof Animal;              // true

Object.create() vs new

// Object.create(proto): creates object with given prototype
const animal = {
  speak() { return `${this.name} makes noise`; }
};
const dog = Object.create(animal);
dog.name = "Rex";
dog.speak(); // "Rex makes noise"

// new calls constructor + sets prototype
function Cat(name) { this.name = name; }
const c = new Cat("Luna");
// Equivalent (roughly) to:
const c2 = Object.create(Cat.prototype);
Cat.call(c2, "Luna");

ES6 Classes (syntactic sugar)

class Shape {
  #area = 0; // private field (ES2022)

  constructor(color) {
    this.color = color;
  }
  describe() { return `A ${this.color} shape`; }
  static create(color) { return new Shape(color); }
}

class Circle extends Shape {
  constructor(color, radius) {
    super(color);        // Must call super() before using this
    this.radius = radius;
  }
  describe() {
    return `${super.describe()} (circle, r=${this.radius})`;
  }
  get area() { return Math.PI * this.radius ** 2; }
  set radius(r) { this._radius = r < 0 ? 0 : r; }
  get radius()  { return this._radius; }
}

const c = new Circle("red", 5);
c.describe();      // "A red shape (circle, r=5)"
c instanceof Circle; // true
c instanceof Shape;  // true

Mixins Pattern

// JS supports single inheritance only — mixins add capabilities
const Serializable = (Base) => class extends Base {
  serialize() { return JSON.stringify(this); }
  static deserialize(json) { return Object.assign(new this(), JSON.parse(json)); }
};

const Validatable = (Base) => class extends Base {
  validate() { return Object.keys(this).every(k => this[k] !== null); }
};

class User extends Serializable(Validatable(class {})) {
  constructor(name, email) {
    super();
    this.name = name;
    this.email = email;
  }
}
const u = new User("Alice", "a@b.com");
u.serialize();  // '{"name":"Alice","email":"a@b.com"}'
u.validate();   // true
1.5 Event Loop

Components

  • Call Stack — LIFO structure for function execution frames
  • Heap — unstructured memory region for object allocation
  • Task Queue (Macrotask) — setTimeout, setInterval, I/O, UI events
  • Microtask Queue — Promise callbacks, queueMicrotask, MutationObserver

Execution Order

1. Run synchronous code to completion (empty call stack)
2. Drain entire microtask queue (including new microtasks added during draining)
3. Render (browser only, between tasks)
4. Pick ONE macrotask, run it
5. Drain microtask queue again → repeat

Classic output question

console.log("1");

setTimeout(() => console.log("2"), 0);

Promise.resolve().then(() => console.log("3"));

console.log("4");

// Output: 1, 4, 3, 2
// 1, 4 → synchronous
// 3    → microtask (Promise) runs before macrotask
// 2    → macrotask (setTimeout)

More complex example

Promise.resolve()
  .then(() => {
    console.log("A");
    setTimeout(() => console.log("B"), 0);
    return Promise.resolve();
  })
  .then(() => console.log("C"));

setTimeout(() => console.log("D"), 0);

// Output: A, C, D, B
// A → first microtask
// C → second microtask (chained .then from "return Promise.resolve()")
// D → first macrotask (registered before B)
// B → second macrotask

requestAnimationFrame

// rAF fires before paint, after current macrotask + microtasks
// Runs at display refresh rate (~60fps = every 16.7ms)
// Paused when tab is hidden → battery efficient

requestAnimationFrame(() => {
  // safe to read/write DOM here
  element.style.transform = `translateX(${x}px)`;
});
1.6 Async Patterns

Callback Hell

// The pyramid of doom
getUser(id, function(err, user) {
  if (err) return handleError(err);
  getOrders(user.id, function(err, orders) {
    if (err) return handleError(err);
    getItems(orders[0].id, function(err, items) {
      if (err) return handleError(err);

Promises

// States: pending → fulfilled | rejected (settled, irreversible)
const p = new Promise((resolve, reject) => {
  setTimeout(() => resolve(42), 1000);
});

p.then(val => console.log(val))   // 42
 .catch(err => console.error(err))
 .finally(() => console.log("done"));

// Chaining — each .then returns a new promise
fetch("/api/user")
  .then(res => res.json())         // returns new promise
  .then(user => user.name)
  .then(name => console.log(name))
  .catch(err => console.error(err)); // catches any error above

async/await

// async function always returns a Promise
// await pauses execution of the async function (not the thread!)

async function fetchUser(id) {
  try {
    const res = await fetch(`/api/users/${id}`);
    if (!res.ok) throw new Error(`HTTP ${res.status}`);
    return await res.json();
  } catch (err) {
    console.error("Failed:", err.message);
    throw err; // re-throw for caller
  }
}

// Sequential vs Parallel
async function sequential() {
  const a = await fetch("/a"); // wait for a
  const b = await fetch("/b"); // then wait for b
  // Total: time(a) + time(b)
}

async function parallel() {
  const [a, b] = await Promise.all([fetch("/a"), fetch("/b")]);
  // Total: max(time(a), time(b))
}

// async/await compiles to:
function fetchUser(id) {
  return fetch(`/api/users/${id}`)
    .then(res => { if(!res.ok) throw new Error(); return res.json(); });
}

Spread / Rest

Generators

function* counter(start = 0) {
  while (true) {
    const reset = yield start++;
    if (reset) start = 0;
  }
}
const gen = counter();
gen.next();        // { value: 0, done: false }
gen.next();        // { value: 1, done: false }
gen.next(true);    // { value: 0, done: false } — reset!

// Finite generator
function* range(start, end, step = 1) {
  for (let i = start; i < end; i += step) yield i;
}
[...range(0, 10, 2)]; // [0, 2, 4, 6, 8]

// Custom iterable with Symbol.iterator
const countdown = {
  [Symbol.iterator]() {
    let n = 3;
    return {
      next() {
        return n > 0
          ? { value: n--, done: false }
          : { value: undefined, done: true };
      }
    };
  }
};
[...countdown]; // [3, 2, 1]

WeakMap, WeakSet, WeakRef

// WeakMap: keys must be objects, not enumerable, allows GC
const cache = new WeakMap();
function process(element) {
  if (cache.has(element)) return cache.get(element);
  const result = expensiveComputation(element);
  cache.set(element, result);
  return result;
}
// When element is GC'd, cache entry is removed automatically

// WeakSet: store objects without preventing GC
const visited = new WeakSet();
function visit(node) {
  if (visited.has(node)) return;
  visited.add(node);
  // process...
}

// WeakRef: hold weak reference
let obj = { data: "big" };
const ref = new WeakRef(obj);
obj = null; // obj can now be GC'd
ref.deref(); // object or undefined if GC'd

Proxy and Reflect

const handler = {

1.8 Functions Deep

Pure Functions & Side Effects

// Pure: same input → same output, no side effects
const add = (a, b) => a + b;          // pure
const double = arr => arr.map(x=>x*2); // pure (returns new array)

// Impure: mutates, reads external state, I/O

Currying

// Manual currying
const curry = fn => {
  const arity = fn.length;
  return function curried(...args) {
    if (args.length >= arity) return fn(...args);
    return (...moreArgs) => curried(...args, ...moreArgs);
  };
};

const add = (a, b, c) => a + b + c;
const curriedAdd = curry(add);
curriedAdd(1)(2)(3);   // 6
curriedAdd(1, 2)(3);   // 6
curriedAdd(1)(2, 3);   // 6

// Real use: configurable functions
const multiply = curry((factor, value) => factor * value);
const triple = multiply(3);
[1, 2, 3].map(triple); // [3, 6, 9]

Function Composition

// compose: right to left
const compose = (...fns) => x => fns.reduceRight((v, f) => f(v), x);

// pipe: left to right
const pipe = (...fns) => x => fns.reduce((v, f) => f(v), x);

const trim = s => s.trim();
const lower = s => s.toLowerCase();
const split = s => s.split(" ");

const process = pipe(trim, lower, split);
process("  Hello World  "); // ["hello", "world"]

Memoization

function memoize(fn) {
  const cache = new Map();
  return function(...args) {
    const key = JSON.stringify(args);
    if (cache.has(key)) return cache.get(key);
    const val = fn.apply(this, args);
    cache.set(key, val);
    return val;
  };
}

// Fibonacci with memoization
const fib = memoize(function(n) {
  if (n <= 1) return n;
  return fib(n - 1) + fib(n - 2);
});
fib(50); // instant instead of exponential

Tail Call Optimization

// Regular recursion: O(n) stack frames
function factorial(n) {
  if (n <= 1) return 1;
  return n * factorial(n - 1); // NOT tail call — multiplies after return
}

// Tail-recursive: tail call is the LAST operation
function factorialTCO(n, acc = 1) {
  if (n <= 1) return acc;
  return factorialTCO(n - 1, n * acc); // tail call — no work after return
}
// ES6 spec mandates TCO in strict mode, but most engines don't implement it
// Prefer iterative for large n
1.9 Memory & Performance

Stack vs Heap

Stack: fixed-size, fast LIFO, stores primitives and function call frames. Automatic deallocation.
Heap: dynamic size, slower, stores objects, arrays, closures. Managed by GC.

Garbage Collection: Mark-and-Sweep

Starting from "roots" (global variables, currently executing functions), the GC marks all reachable objects. Everything unreachable is swept (freed). Modern V8 uses generational GC: young generation (Minor GC, fast) and old generation (Major GC, slower).

Memory Leaks

// 1. Forgotten timers
function startPolling() {
  setInterval(() => {
    fetchData(); // interval keeps running even if component unmounts
  }, 1000);
}
// Fix: return clearInterval() in useEffect cleanup

// 2. Closures holding large refs
function processLargeData() {
  const HUGE = new Array(1e6).fill("data");
  return function leaky() {
    // Only uses HUGE.length but holds entire array in memory
    return HUGE.length;

// Fix: capture only what you need
function processLargeDataFixed() {
  const HUGE = new Array(1e6).fill("data");
  const len = HUGE.length;
  return function() { return len; }; // HUGE can be GC'd

⚡ Key Takeaways

  • box-sizing: border-box makes width/height include padding and border — apply it globally in resets

3.1 Core Concepts

JSX Compilation

// JSX is syntactic sugar for React.createElement (classic runtime)
// or just _jsx (new automatic runtime — React 17+)

// Before (classic):

// Compiles to:

Virtual DOM & Reconciliation

React maintains a virtual DOM (lightweight JS representation). When state changes, React creates a new vDOM tree and diffs it against the previous one (reconciliation). Only the minimal set of real DOM changes are applied (commit phase).

Fiber Architecture

React Fiber (React 16+) is a rewrite of the reconciliation engine. Each component has a "fiber" — a JS object with component info, pending updates, and links to parent/child/sibling fibers. Fiber enables:

  • Incremental rendering — split work into chunks, pause and resume
  • Prioritization — urgent updates (typing) beat non-urgent (data fetch)
  • Concurrent features — startTransition, Suspense

useState

const [state, setState] = useState(initialValue);

  setTimeout(() => {

// Fix: functional update

useEffect(() => {

4.1 Architecture

Old Architecture

JavaScript Thread → Bridge (async, JSON-serialized messages) → Native Thread. The bridge was a bottleneck: async only, serialization overhead, can't share memory.

New Architecture (React Native 0.68+)

  • JSI (JavaScript Interface): Direct C++ binding. JS can hold references to native objects and call methods synchronously without serialization.
  • Fabric: New rendering system. C++ shadow tree, synchronous layout measurement, concurrent features support.
  • TurboModules: Lazy-loaded native modules via JSI. Only loaded when first used.
  • Codegen: TypeScript/Flow specs generate native code automatically.

Hermes Engine

Bytecode-compiled JS engine optimized for React Native. Faster startup (pre-compiled), lower memory, built-in debugger. Default since RN 0.70.

Yoga Layout Engine

Cross-platform C++ Flexbox implementation used to calculate layouts. Runs on native thread. Implements a subset of CSS Flexbox.

5.1 Event Loop (Node-specific)

Phases of the Node.js Event Loop

// 1. timers:          setTimeout, setInterval callbacks
// 2. pending:         I/O callbacks from previous iteration
// 3. idle/prepare:    internal use
// 4. poll:            retrieve new I/O events (blocking if queue empty)
// 5. check:           setImmediate callbacks
// 6. close callbacks: socket.on("close"), etc.

// Between each phase: process.nextTick + Promise microtasks drain

setImmediate vs setTimeout(fn, 0)

// In the main module: order is non-deterministic (system/timing dependent)

setImmediate(() => console.log("immediate"));
// May output either order

// Inside I/O callback: setImmediate ALWAYS runs first

fs.readFile("file.txt", () => {
  setTimeout(() => console.log("timeout"), 0);
  setImmediate(() => console.log("immediate")); // always first
});

// process.nextTick: runs BEFORE next event loop phase (before I/O)
process.nextTick(() => console.log("nextTick"));
Promise.resolve().then(() => console.log("promise"));
setImmediate(() => console.log("setImmediate"));
setTimeout(() => console.log("setTimeout"), 0);

Circuit Breaker Pattern

// CLOSED: requests pass through, monitoring error rate
// OPEN: all requests fail fast after threshold exceeded
// HALF-OPEN: probe request; success closes, failure reopens

class CircuitBreaker {
  constructor(fn, { threshold = 5, timeout = 60000 } = {}) {
    this.fn = fn; this.state = "CLOSED"; this.failures = 0;
    this.threshold = threshold; this.timeout = timeout;
  }
  async call(...args) {
    if (this.state === "OPEN") throw new Error("Circuit OPEN");
    try {
      const result = await this.fn(...args);
      this.failures = 0; this.state = "CLOSED";
      return result;
    } catch (err) {
      this.failures++;
      if (this.failures >= this.threshold) {
        this.state = "OPEN";
        setTimeout(() => { this.state = "HALF-OPEN"; }, this.timeout);
      }
      throw err;
    }
  }
}

// Violation: one class does too many things
class UserService {
  saveUser(user) { db.save(user); }
  sendWelcomeEmail(user) { emailClient.send(user.email, "Welcome!"); }
  generateReport() { return users.map(u => u.name).join("\n"); }
}
// Fixed: separate concerns into focused classes
class UserRepository { save(user) { db.save(user); } }
class EmailService   { sendWelcome(user) { emailClient.send(user.email, "Welcome!"); } }
class ReportService  { generate(users) { return users.map(u => u.name).join("\n"); } }

O — Open/Closed

// Open for extension, closed for modification
// Violation: must edit AreaCalculator to add new shapes
class AreaCalculator {
  calculate(shape) {
    if (shape.type === "circle") return Math.PI * shape.r ** 2;
    if (shape.type === "square") return shape.side ** 2;
  }
}
// Fixed: each shape knows its area
class Circle    { area() { return Math.PI * this.r ** 2; } }
class Square    { area() { return this.side ** 2; } }
class Triangle  { area() { return 0.5 * this.base * this.height; } }
class AreaCalculator { calculate(shape) { return shape.area(); } }

L — Liskov Substitution

// Subtypes must be substitutable for base types without altering correctness
// Violation: Square extends Rectangle but breaks invariant
class Rectangle {
  setWidth(w)  { this.w = w; }
  setHeight(h) { this.h = h; }
  area() { return this.w * this.h; }
}
class Square extends Rectangle {
  setWidth(w)  { this.w = this.h = w; } // breaks: caller expects independent dims
  setHeight(h) { this.w = this.h = h; }
}
// Caller sets width=5, height=10, expects area=50 — Square returns 100

I — Interface Segregation

// Don't force implementations of unused methods
// Violation: Robot forced to implement eat()
interface Worker { work(): void; eat(): void; sleep(): void; }
class Robot implements Worker {
  work() {}
  eat() { throw new Error("Robots don't eat!"); }
  sleep() { throw new Error("Robots don't sleep!"); }
}
// Fixed: smaller interfaces
interface Workable { work(): void; }
interface Biological { eat(): void; sleep(): void; }
class Human implements Workable, Biological { work(){} eat(){} sleep(){} }
class Robot implements Workable { work(){} }

D — Dependency Inversion

// High-level modules should not depend on low-level modules
// Violation: tight coupling to MySQL
class UserService {
  constructor() { this.db = new MySQLDatabase(); }
}
// Fixed: inject dependency via abstraction
interface Database { findById(id: string): Promise<User>; }
class UserService {
  constructor(private db: Database) {} // inject
}
const service = new UserService(new PostgreSQLDatabase());
// Easy to swap implementations and mock in tests
9.2 Other Principles
  • DRY (Don't Repeat Yourself): abstract duplicate code. But don't over-abstract prematurely — wrong DRY creates tight coupling.
  • KISS (Keep It Simple): prefer simple solutions. Complexity = bugs + maintenance burden.
  • YAGNI (You Aren't Gonna Need It): don't build features until actually needed. Avoid speculative generality.
  • Law of Demeter: talk only to immediate friends. Avoid a.getB().getC().doWork(). Prefer a.doWork() which coordinates internally. Reduces coupling.

// Factory Method: create objects without specifying class
class NotificationFactory {
  static create(type) {
    const types = { email: EmailNotification, sms: SMSNotification, push: PushNotification };
    if (!types[type]) throw new Error(`Unknown: ${type}`);
    return new types[type]();
  }

// Builder: construct complex objects step by step
class QueryBuilder {
  #parts = { table: "", conditions: [], limit: null, order: null };
  from(t)   { this.#parts.table = t; return this; }
  where(c)  { this.#parts.conditions.push(c); return this; }
  limit(n)  { this.#parts.limit = n; return this; }
  orderBy(c){ this.#parts.order = c; return this; }
  build() {
    let sql = `SELECT * FROM ${this.#parts.table}`;
    if (this.#parts.conditions.length) sql += ` WHERE ${this.#parts.conditions.join(" AND ")}`;
    if (this.#parts.order)  sql += ` ORDER BY ${this.#parts.order}`;
    if (this.#parts.limit)  sql += ` LIMIT ${this.#parts.limit}`;
    return sql;
  }
}
const q = new QueryBuilder().from("users").where("active = true").limit(10).build();

// Observer / EventBus
class EventBus {
  #listeners = new Map();
  on(event, handler) {
    if (!this.#listeners.has(event)) this.#listeners.set(event, []);
    this.#listeners.get(event).push(handler);
    return () => this.off(event, handler);  // returns unsubscribe function
  }
  off(event, handler) {
    this.#listeners.set(event, (this.#listeners.get(event) || []).filter(h => h !== handler));
  }
  emit(event, data) { (this.#listeners.get(event) || []).forEach(h => h(data)); }
}

// Strategy: interchangeable algorithms

  constructor(strategy) { this.strategy = strategy; }
  sort(data) { return this.strategy(data); }
}

const descending = new Sorter(arr => [...arr].sort((a,b) => b-a));

  undo() { this.#history.pop()?.undo(); }

  handle(req) { return this.next?.handle(req); }

    return super.handle(req);

}

  • SOLID: each principle addresses a different failure mode — don't memorize definitions, memorize the problems they prevent
  • Dependency Inversion: high-level modules depend on abstractions, not implementations — enables testability via mocks

10.2 Tricky TypeScript Questions

Common gotchas

// Q: Why does narrowing not work inside callbacks?
let x: string | null = "hello";
setTimeout(() => {
  x.toUpperCase(); // Error: x might be null (could change before callback)
}, 0);
// Fix: capture value
const copy = x;
setTimeout(() => copy?.toUpperCase(), 0);

// Q: Structural typing — same structure = compatible
interface Cat { meow(): void; }
interface Dog { meow(): void; } // same shape
const cat: Cat = { meow() {} };
const dog: Dog = cat; // OK! TS uses structural (not nominal) typing

// Q: Type widening with const
let x = 3;        // type: number (widened)
const y = 3;      // type: 3 (literal, not widened)
const z = [1,2,3] as const; // type: readonly [1, 2, 3]

// Q: keyof typeof
const obj = { a: 1, b: 2, c: 3 };
type Keys = keyof typeof obj; // "a" | "b" | "c"
function getValue(o: typeof obj, key: Keys) { return o[key]; }

// Q: Conditional distribution
type ToArray<T> = T extends any ? T[] : never;
type Result = ToArray<string | number>; // string[] | number[] (distributes)
// vs:
type NoDistrib<T> = [T] extends [any] ? T[] : never;
type Result2 = NoDistrib<string | number>; // (string | number)[]

⚡ Key Takeaways

  • Structural typing: two types are compatible if their shapes match, regardless of name — duck typing with static analysis
  • Generics + constraints (extends) make code reusable without sacrificing type safety
  • Conditional types (T extends U ? X : Y) and mapped types ({ [K in keyof T]: ... }) enable powerful type transformations
  • unknown is safer than any — you must narrow it before use; any opts out of all type checking
  • as const freezes types to literal values, enabling precise discriminated unions
  • Utility types (Partial, Pick, Omit, ReturnType) eliminate boilerplate type definitions

12.1 Core Git Concepts & Commands

Merge vs Rebase

# Merge: creates merge commit, preserves history
git checkout main
git merge feature/login  # merge commit created, diverge is visible in log

# Rebase: moves feature commits to tip of target, linear history
git checkout feature/login
git rebase main  # replays feature commits on top of current main

# Golden Rule of Rebase:
# NEVER rebase public/shared branches (rewrites history = new commit SHAs)
# Safe: rebase local feature branches before creating PR

# Interactive rebase: rewrite last N commits
git rebase -i HEAD~4
# pick   = use commit as-is
# reword = change commit message
# edit   = pause to amend commit (git commit --amend, then git rebase --continue)
# squash = combine with previous commit (keep both messages)
# fixup  = combine, discard this commit's message
# drop   = remove commit entirely

Useful Commands

# Stash: temporarily save uncommitted work
git stash push -m "WIP: login feature"
git stash list                   # list all stashes
git stash pop                    # apply latest and remove
git stash apply stash@{2}        # apply specific, keep in stash
git stash drop stash@{0}         # delete specific stash

# Cherry-pick: apply specific commit(s) to current branch
git cherry-pick abc123
git cherry-pick abc123^..def456  # range (inclusive)

# Bisect: binary search for bug-introducing commit
git bisect start
git bisect bad HEAD              # current commit is bad
git bisect good v1.2.0           # last known good tag
# Git checks out middle commit, you test, then:
git bisect good  # or bad
# Repeat until Git reports the first bad commit
git bisect reset

# Reflog: safety net — all HEAD movements
git reflog
git checkout HEAD@{5}            # restore to 5 HEAD moves ago
git reset --hard HEAD@{3}        # undo to 3 reflog entries ago

13.4 Service Workers & Web Workers 
// Service Worker: intercept network requests, offline caching
self.addEventListener("install", event => {
  event.waitUntil(
    caches.open("v1").then(cache => cache.addAll(["/", "/app.js", "/style.css"]))
  );
});

self.addEventListener("fetch", event => {
  event.respondWith(
    caches.match(event.request).then(cached => {
      const network = fetch(event.request).then(response => {
        caches.open("v1").then(c => c.put(event.request, response.clone()));
        return response;
      });
      return cached || network;  // cache-first
    })
  );
});

14.3 Authentication Best Practices 
// Password hashing: adaptive algorithms that are slow by design
import bcrypt from "bcrypt";
const saltRounds = 12; // 2^12 iterations, ~200ms — increase over time
const hash = await bcrypt.hash(password, saltRounds);
const isValid = await bcrypt.compare(password, hash);

// argon2: winner of Password Hashing Competition (preferred over bcrypt)
import argon2 from "argon2";
const hash = await argon2.hash(password, {
  type: argon2.argon2id,  // combines argon2i + argon2d — best for passwords
  memoryCost: 65536,      // 64 MB RAM required
  timeCost: 3,            // iterations
  parallelism: 4          // threads
});
const isValid = await argon2.verify(hash, password);

// NEVER use: MD5, SHA1, SHA256 without salt for passwords (too fast — GPU brute-force)

// JWT security checklist
// 1. Short-lived access tokens: 15 minutes

⚡ Key Takeaways

  • XSS: sanitize all user input and use Content Security Policy; innerHTML with untrusted data is dangerous
  • CSRF: use SameSite=Strict cookies and server-side CSRF tokens for all state-changing requests
  • SQL injection: always use parameterized queries — never concatenate user input into SQL strings

  • Rotate secrets regularly; never commit secrets to git — use environment variables and a secrets manager
  • npm audit and Dependabot catch known CVEs in dependencies; treat dependency updates as regular maintenance
