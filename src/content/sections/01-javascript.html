1. JavaScript (Deep) 11 topics

JavaScript is the most-tested language in frontend interviews. This section covers the subtle runtime behaviors — coercion, closures, the event loop, prototype chains, and async patterns — that interviewers use to distinguish strong candidates. Expect at least one deep JS question in any frontend or fullstack interview.

1.1 Core Types & Coercion

Primitive Types

JavaScript has 8 types: string, number, bigint, boolean, undefined, null, symbol, and object (everything else). The first 7 are primitives — immutable and compared by value.

typeof {}             // "object"
typeof []             // "object"  ← arrays are objects
typeof function(){}   // "function" ← special case
typeof Symbol()       // "symbol"

1.2 Scope & Closures

Lexical vs Dynamic Scope

JavaScript uses lexical (static) scope — where a function is defined determines what variables it can access, not where it's called. Dynamic scope (as in Bash) would use the call site instead.

Hoisting

// var: declaration hoisted, initialization is undefined
console.log(x); // undefined (not ReferenceError)
var x = 5;
// Equivalent to: var x; console.log(x); x = 5;

// let/const: declaration hoisted but NOT initialized → TDZ
console.log(y); // ReferenceError: Cannot access 'y' before initialization
let y = 5;
// The Temporal Dead Zone (TDZ) is from block entry to the declaration

// Function declarations: fully hoisted (both name and body)
greet(); // "hello" — works!
function greet() { console.log("hello"); }

// Function expressions: only var is hoisted (as undefined)
sayHi(); // TypeError: sayHi is not a function
var sayHi = function() { console.log("hi"); };

// Arrow functions: same as function expressions
go(); // TypeError
var go = () => {};

Closures

A closure is a function that remembers the variables from its outer scope even after the outer function has returned.

// Classic closure: counter
function makeCounter(start = 0) {
  let count = start;
  return {
    increment() { return ++count; },
    decrement() { return --count; },
    value()     { return count; }
  };
}
const c = makeCounter(10);
c.increment(); // 11
c.increment(); // 12
c.value();     // 12

// Memoization via closure
function memoize(fn) {
  const cache = new Map();
  return function(...args) {
    const key = JSON.stringify(args);
    if (cache.has(key)) return cache.get(key);
    const result = fn.apply(this, args);
    cache.set(key, result);
    return result;
  };
}

// Partial application via closure
function multiply(a, b) { return a * b; }
const double = memoize(multiply.bind(null, 2));
double(5); // 10

// Module pattern
const BankAccount = (function() {
  let balance = 0; // private
  return {
    deposit(amt)  { balance += amt; },
    withdraw(amt) { if (amt <= balance) balance -= amt; },
    getBalance()  { return balance; }
  };
})();

IIFE (Immediately Invoked Function Expression)

(function() {
  var private = "not polluting global scope";
  console.log(private);
})();

// Arrow IIFE
(() => {
  const x = 42;
})();

let in Loops — Classic Interview Trap

// var: all callbacks share the SAME variable
for (var i = 0; i < 3; i++) {
  setTimeout(() => console.log(i), 0);
}
// Output: 3 3 3  (i is 3 by the time callbacks run)

// let: each iteration gets its OWN binding
for (let i = 0; i < 3; i++) {
  setTimeout(() => console.log(i), 0);
}
// Output: 0 1 2  ✓

// var workaround with IIFE
for (var i = 0; i < 3; i++) {
  (function(j) {
    setTimeout(() => console.log(j), 0);
  })(i);
}
// Output: 0 1 2
1.3 this & Context

The 4 Rules of this (in order of precedence)

1. new binding

function Person(name) { this.name = name; }
const p = new Person("Alice");
// 1. new object created
// 2. [[Prototype]] linked
// 3. this bound to new object
// 4. object returned (unless constructor returns object)

2. Explicit binding (call/apply/bind)

function greet() { return `Hello, ${this.name}`; }
const obj = { name: "Bob" };

greet.call(obj);           // "Hello, Bob"  — call(thisArg, arg1, arg2)
greet.apply(obj);          // "Hello, Bob"  — apply(thisArg, [args])
const boundGreet = greet.bind(obj);
boundGreet();              // "Hello, Bob"  — returns new function

// bind vs call vs apply
Math.max.call(null, 1, 2, 3);    // 3
Math.max.apply(null, [1, 2, 3]); // 3  — useful when args are in array

// Partial application with bind
function add(a, b) { return a + b; }
const add5 = add.bind(null, 5);
add5(3); // 8

3. Implicit binding

const obj = {
  name: "Carol",
  greet() { return `Hello, ${this.name}`; }
};
obj.greet(); // "Hello, Carol" — this = obj (call site has context object)

// Losing implicit binding
const fn = obj.greet;
fn(); // "Hello, undefined" — this = global (or undefined in strict mode)

// Fix: bind
const fn2 = obj.greet.bind(obj);
fn2(); // "Hello, Carol"

// Losing this in callbacks
setTimeout(obj.greet, 100);         // broken
setTimeout(obj.greet.bind(obj), 100); // fixed
setTimeout(() => obj.greet(), 100);   // also fixed

4. Default binding

function whoAmI() { return this; }
whoAmI();          // window (sloppy mode) or undefined (strict mode)
// "use strict" at top of file or function prevents default binding

Arrow Functions: Lexical this

class Timer {
  constructor() {
    this.seconds = 0;
  }
  start() {
    // Arrow function: inherits this from start() method's this
    setInterval(() => {
      this.seconds++;
      console.log(this.seconds);
    }, 1000);
  }
}

// vs. regular function (broken):
class BrokenTimer {
  start() {
    setInterval(function() {
      this.seconds++; // this is undefined in strict mode!
    }, 1000);
  }
}

// Arrow functions CANNOT be used as constructors
const Foo = () => {};
new Foo(); // TypeError: Foo is not a constructor

// Arrow functions have no 'arguments' object
const f = () => { console.log(arguments); }; // ReferenceError

this in Classes

class Animal {
  constructor(name) {
    this.name = name;
  }
  speak() {
    return `${this.name} makes a noise.`;
  }
}
class Dog extends Animal {
  speak() {
    return `${this.name} barks.`;
  }
}
const d = new Dog("Rex");
d.speak(); // "Rex barks."

// Class methods lose this when passed as callbacks
const { speak } = d;
speak(); // TypeError or undefined name (strict mode)
1.4 Prototypes & Inheritance

The Prototype Chain

// Every object has [[Prototype]] (exposed as __proto__, non-standard)
const arr = [1, 2, 3];
// arr → Array.prototype → Object.prototype → null

// Functions have a .prototype property used for inheritance
function Animal(name) { this.name = name; }
Animal.prototype.speak = function() { return this.name; };

const dog = new Animal("Rex");
dog.speak();                        // "Rex" — found on Animal.prototype
dog.hasOwnProperty("name");         // true
dog.hasOwnProperty("speak");        // false — inherited
Object.getPrototypeOf(dog) === Animal.prototype; // true
dog instanceof Animal;              // true

Object.create() vs new

// Object.create(proto): creates object with given prototype
const animal = {
  speak() { return `${this.name} makes noise`; }
};
const dog = Object.create(animal);
dog.name = "Rex";
dog.speak(); // "Rex makes noise"

// new calls constructor + sets prototype
function Cat(name) { this.name = name; }
const c = new Cat("Luna");
// Equivalent (roughly) to:
const c2 = Object.create(Cat.prototype);
Cat.call(c2, "Luna");

ES6 Classes (syntactic sugar)

class Shape {
  #area = 0; // private field (ES2022)

  constructor(color) {
    this.color = color;
  }
  describe() { return `A ${this.color} shape`; }
  static create(color) { return new Shape(color); }
}

class Circle extends Shape {
  constructor(color, radius) {
    super(color);        // Must call super() before using this
    this.radius = radius;
  }
  describe() {
    return `${super.describe()} (circle, r=${this.radius})`;
  }
  get area() { return Math.PI * this.radius ** 2; }
  set radius(r) { this._radius = r < 0 ? 0 : r; }
  get radius()  { return this._radius; }
}

const c = new Circle("red", 5);
c.describe();      // "A red shape (circle, r=5)"
c instanceof Circle; // true
c instanceof Shape;  // true

Mixins Pattern

// JS supports single inheritance only — mixins add capabilities
const Serializable = (Base) => class extends Base {
  serialize() { return JSON.stringify(this); }
  static deserialize(json) { return Object.assign(new this(), JSON.parse(json)); }
};

const Validatable = (Base) => class extends Base {
  validate() { return Object.keys(this).every(k => this[k] !== null); }
};

class User extends Serializable(Validatable(class {})) {
  constructor(name, email) {
    super();
    this.name = name;
    this.email = email;
  }
}
const u = new User("Alice", "a@b.com");
u.serialize();  // '{"name":"Alice","email":"a@b.com"}'
u.validate();   // true
1.5 Event Loop

Components

  • Call Stack — LIFO structure for function execution frames
  • Heap — unstructured memory region for object allocation
  • Task Queue (Macrotask) — setTimeout, setInterval, I/O, UI events
  • Microtask Queue — Promise callbacks, queueMicrotask, MutationObserver

Execution Order

1. Run synchronous code to completion (empty call stack)
2. Drain entire microtask queue (including new microtasks added during draining)
3. Render (browser only, between tasks)
4. Pick ONE macrotask, run it
5. Drain microtask queue again → repeat

Classic output question

console.log("1");

setTimeout(() => console.log("2"), 0);

Promise.resolve().then(() => console.log("3"));

console.log("4");

// Output: 1, 4, 3, 2
// 1, 4 → synchronous
// 3    → microtask (Promise) runs before macrotask
// 2    → macrotask (setTimeout)

More complex example

Promise.resolve()
  .then(() => {
    console.log("A");
    setTimeout(() => console.log("B"), 0);
    return Promise.resolve();
  })
  .then(() => console.log("C"));

setTimeout(() => console.log("D"), 0);

// Output: A, C, D, B
// A → first microtask
// C → second microtask (chained .then from "return Promise.resolve()")
// D → first macrotask (registered before B)
// B → second macrotask

requestAnimationFrame

// rAF fires before paint, after current macrotask + microtasks
// Runs at display refresh rate (~60fps = every 16.7ms)
// Paused when tab is hidden → battery efficient

requestAnimationFrame(() => {
  // safe to read/write DOM here
  element.style.transform = `translateX(${x}px)`;
});
1.6 Async Patterns

Callback Hell

// The pyramid of doom
getUser(id, function(err, user) {
  if (err) return handleError(err);
  getOrders(user.id, function(err, orders) {
    if (err) return handleError(err);
    getItems(orders[0].id, function(err, items) {
      if (err) return handleError(err);
      // ... even deeper
    });
  });
});

Promises

// States: pending → fulfilled | rejected (settled, irreversible)
const p = new Promise((resolve, reject) => {
  setTimeout(() => resolve(42), 1000);
});

p.then(val => console.log(val))   // 42
 .catch(err => console.error(err))
 .finally(() => console.log("done"));

// Chaining — each .then returns a new promise
fetch("/api/user")
  .then(res => res.json())         // returns new promise
  .then(user => user.name)
  .then(name => console.log(name))
  .catch(err => console.error(err)); // catches any error above

Promise combinators comparison

Promise.allALL fulfillANY rejectsarray of values
Promise.allSettledALL settleneverarray of {status, value/reason}
Promise.raceFIRST settlesFIRST rejectssingle value/reason

async/await

// async function always returns a Promise
// await pauses execution of the async function (not the thread!)

async function fetchUser(id) {
  try {
    const res = await fetch(`/api/users/${id}`);
    if (!res.ok) throw new Error(`HTTP ${res.status}`);
    return await res.json();
  } catch (err) {
    console.error("Failed:", err.message);
    throw err; // re-throw for caller
  }
}

// Sequential vs Parallel
async function sequential() {
  const a = await fetch("/a"); // wait for a
  const b = await fetch("/b"); // then wait for b
  // Total: time(a) + time(b)
}

async function parallel() {
  const [a, b] = await Promise.all([fetch("/a"), fetch("/b")]);
  // Total: max(time(a), time(b))
}

// async/await compiles to:
function fetchUser(id) {
  return fetch(`/api/users/${id}`)
    .then(res => { if(!res.ok) throw new Error(); return res.json(); });
}

AbortController

const controller = new AbortController();
const { signal } = controller;

fetch("/api/large-data", { signal })
  .then(res => res.json())
  .catch(err => {
    if (err.name === "AbortError") console.log("Request aborted");
    else throw err;
  });

// Cancel after 5 seconds
setTimeout(() => controller.abort(), 5000);

// React cleanup pattern
useEffect(() => {
  const controller = new AbortController();
  fetch("/api/data", { signal: controller.signal })
    .then(r => r.json())
    .then(setData)
    .catch(() => {});
  return () => controller.abort();
}, []);

Destructuring

// Array destructuring

// a=1, b=2, rest=[3,4,5]

// userName = "Bob"

// Nested

function greet({ name = "stranger", age = 0 } = {}) {
  return `${name} is ${age}`;

Spread / Rest

// Spread in arrays
const arr1 = [1, 2], arr2 = [3, 4];
const merged = [...arr1, ...arr2]; // [1,2,3,4]

// Spread in objects (shallow clone)
const obj1 = { a: 1 };
const obj2 = { ...obj1, b: 2 }; // { a:1, b:2 }
const override = { ...obj1, a: 99 }; // { a:99 }

// Rest in functions
function sum(...nums) { return nums.reduce((a, b) => a + b, 0); }

// Spread to pass array as args
Math.max(...[1, 5, 3]); // 5

Generators

function* counter(start = 0) {
  while (true) {
    const reset = yield start++;
    if (reset) start = 0;
  }
}
const gen = counter();
gen.next();        // { value: 0, done: false }
gen.next();        // { value: 1, done: false }
gen.next(true);    // { value: 0, done: false } — reset!

// Finite generator
function* range(start, end, step = 1) {
  for (let i = start; i < end; i += step) yield i;
}
[...range(0, 10, 2)]; // [0, 2, 4, 6, 8]

// Custom iterable with Symbol.iterator
const countdown = {
  [Symbol.iterator]() {
    let n = 3;
    return {
      next() {
        return n > 0
          ? { value: n--, done: false }
          : { value: undefined, done: true };
      }
    };
  }
};
[...countdown]; // [3, 2, 1]

WeakMap, WeakSet, WeakRef

// WeakMap: keys must be objects, not enumerable, allows GC
const cache = new WeakMap();
function process(element) {
  if (cache.has(element)) return cache.get(element);
  const result = expensiveComputation(element);
  cache.set(element, result);
  return result;
}
// When element is GC'd, cache entry is removed automatically

// WeakSet: store objects without preventing GC
const visited = new WeakSet();
function visit(node) {
  if (visited.has(node)) return;
  visited.add(node);
  // process...
}

// WeakRef: hold weak reference
let obj = { data: "big" };
const ref = new WeakRef(obj);
obj = null; // obj can now be GC'd
ref.deref(); // object or undefined if GC'd

Proxy and Reflect

const handler = {
  get(target, prop, receiver) {
    console.log(`Getting ${prop}`);
    return Reflect.get(target, prop, receiver);
  },
  set(target, prop, value, receiver) {
    if (typeof value !== "number") throw new TypeError("Must be number");
    return Reflect.set(target, prop, value, receiver);
  },
  has(target, prop) {
    console.log(`Checking ${prop}`);
    return prop in target;
  }
};

const data = new Proxy({ x: 1 }, handler);
data.x;    // logs "Getting x"
data.y = 5;  // works
data.y = "a"; // TypeError

// Use cases: validation, logging, virtual properties, reactive systems

Optional Chaining, Nullish Coalescing, Logical Assignment

// Optional chaining ?.
const city = user?.address?.city;   // undefined instead of TypeError
const first = arr?.[0];             // safe array access
const result = obj?.method?.();     // safe method call

// Nullish coalescing: only null/undefined (not 0, "")
const port = config.port ?? 3000;

// vs OR (problematic for falsy values)
const port2 = config.port || 3000;  // 0 → 3000 (wrong!)

// Logical assignment
x ??= defaultValue;  // x = x ?? defaultValue
x &&= transform(x);  // x = x && transform(x)
x ||= fallback;      // x = x || fallback

// structuredClone — deep clone

1.8 Functions Deep

Pure Functions & Side Effects

// Pure: same input → same output, no side effects
const add = (a, b) => a + b;          // pure
const double = arr => arr.map(x=>x*2); // pure (returns new array)

// Impure: mutates, reads external state, I/O
let count = 0;
const increment = () => ++count;       // impure (mutates external state)
const getTime = () => Date.now();      // impure (non-deterministic)

Currying

// Manual currying
const curry = fn => {
  const arity = fn.length;
  return function curried(...args) {
    if (args.length >= arity) return fn(...args);
    return (...moreArgs) => curried(...args, ...moreArgs);
  };
};

const add = (a, b, c) => a + b + c;
const curriedAdd = curry(add);
curriedAdd(1)(2)(3);   // 6
curriedAdd(1, 2)(3);   // 6
curriedAdd(1)(2, 3);   // 6

// Real use: configurable functions
const multiply = curry((factor, value) => factor * value);
const triple = multiply(3);
[1, 2, 3].map(triple); // [3, 6, 9]

Function Composition

// compose: right to left
const compose = (...fns) => x => fns.reduceRight((v, f) => f(v), x);

// pipe: left to right
const pipe = (...fns) => x => fns.reduce((v, f) => f(v), x);

const trim = s => s.trim();
const lower = s => s.toLowerCase();
const split = s => s.split(" ");

const process = pipe(trim, lower, split);
process("  Hello World  "); // ["hello", "world"]

Memoization

function memoize(fn) {
  const cache = new Map();
  return function(...args) {
    const key = JSON.stringify(args);
    if (cache.has(key)) return cache.get(key);
    const val = fn.apply(this, args);
    cache.set(key, val);
    return val;
  };
}

// Fibonacci with memoization
const fib = memoize(function(n) {
  if (n <= 1) return n;
  return fib(n - 1) + fib(n - 2);
});
fib(50); // instant instead of exponential

Tail Call Optimization

// Regular recursion: O(n) stack frames
function factorial(n) {
  if (n <= 1) return 1;
  return n * factorial(n - 1); // NOT tail call — multiplies after return
}

// Tail-recursive: tail call is the LAST operation
function factorialTCO(n, acc = 1) {
  if (n <= 1) return acc;
  return factorialTCO(n - 1, n * acc); // tail call — no work after return
}
// ES6 spec mandates TCO in strict mode, but most engines don't implement it
// Prefer iterative for large n
1.9 Memory & Performance

Stack vs Heap

Stack: fixed-size, fast LIFO, stores primitives and function call frames. Automatic deallocation.
Heap: dynamic size, slower, stores objects, arrays, closures. Managed by GC.

Garbage Collection: Mark-and-Sweep

Starting from "roots" (global variables, currently executing functions), the GC marks all reachable objects. Everything unreachable is swept (freed). Modern V8 uses generational GC: young generation (Minor GC, fast) and old generation (Major GC, slower).

Memory Leaks

// 1. Forgotten timers
function startPolling() {
  setInterval(() => {
    fetchData(); // interval keeps running even if component unmounts
  }, 1000);
}
// Fix: return clearInterval() in useEffect cleanup

// 2. Closures holding large refs
function processLargeData() {
  const HUGE = new Array(1e6).fill("data");
  return function leaky() {
    // Only uses HUGE.length but holds entire array in memory
    return HUGE.length;
  };
}
// Fix: capture only what you need
function processLargeDataFixed() {
  const HUGE = new Array(1e6).fill("data");
  const len = HUGE.length;
  return function() { return len; }; // HUGE can be GC'd
}

// 3. Detached DOM nodes
let button = document.getElementById("btn");
const handler = () => console.log("clicked");
button.addEventListener("click", handler);
document.body.removeChild(button);
// button still referenced → not GC'd
// Fix:
button.removeEventListener("click", handler);
button = null;

// 4. Global variables accidentally created
function oops() {
  leak = "I am global now"; // missing var/let/const
}

// 5. Event emitters without cleanup
emitter.on("data", handler);
// Fix: emitter.off("data", handler) when done

⚡ Key Takeaways

  • typeof null === "object" is a historical bug — always use === null to check for null
  • NaN is the only value not equal to itself; use Number.isNaN(), not isNaN()
  • Closures capture variable references, not values — the classic loop var bug stems from this
  • Event loop order: call stack → microtasks (Promises, queueMicrotask) → macrotasks (setTimeout, I/O)
  • == performs type coercion via the Abstract Equality algorithm; prefer ===
  • Prototype chain walks up __proto__ until null; hasOwnProperty checks own properties only
  • async/await is syntactic sugar over Promises; errors must be caught with try/catch
  • Object.is() distinguishes -0 from +0 and treats NaN === NaN
