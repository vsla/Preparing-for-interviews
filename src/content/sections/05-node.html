5. Node.js (Deep) 7 topics

Node.js interviews test your understanding of single-threaded non-blocking I/O. The event loop phases, stream backpressure, cluster vs worker threads, and production patterns like graceful shutdown are common targets. Knowing libuv internals at a conceptual level is a strong differentiator.

5.1 Event Loop (Node-specific)

Phases of the Node.js Event Loop

// 1. timers:          setTimeout, setInterval callbacks
// 2. pending:         I/O callbacks from previous iteration
// 3. idle/prepare:    internal use
// 4. poll:            retrieve new I/O events (blocking if queue empty)
// 5. check:           setImmediate callbacks
// 6. close callbacks: socket.on("close"), etc.

// Between each phase: process.nextTick + Promise microtasks drain

setImmediate vs setTimeout(fn, 0)

// In the main module: order is non-deterministic (system/timing dependent)
setTimeout(() => console.log("timeout"), 0);
setImmediate(() => console.log("immediate"));
// May output either order

// Inside I/O callback: setImmediate ALWAYS runs first
const fs = require("fs");
fs.readFile("file.txt", () => {
  setTimeout(() => console.log("timeout"), 0);
  setImmediate(() => console.log("immediate")); // always first
});

// process.nextTick: runs BEFORE next event loop phase (before I/O)
process.nextTick(() => console.log("nextTick"));
Promise.resolve().then(() => console.log("promise"));
setImmediate(() => console.log("setImmediate"));
setTimeout(() => console.log("setTimeout"), 0);

// Output: nextTick, promise, setTimeout/setImmediate (order varies), setImmediate

// libuv thread pool (default 4 threads, UV_THREADPOOL_SIZE)
// Used by: fs (most ops), crypto (pbkdf2, randomBytes), dns.lookup, zlib

5.3 Streams 
const { Readable, Writable, Transform, pipeline } = require("stream");

// Why streams: process data in chunks instead of loading entirely into memory
// 1GB file: buffering needs 1GB RAM; streaming needs ~64KB

// Readable stream
const { createReadStream } = require("fs");
const readable = createReadStream("big.file");
readable.on("data", chunk => process(chunk));
readable.on("end", () => console.log("done"));
readable.on("error", err => console.error(err));

// Backpressure: writable signals it's full → readable pauses
// pipe handles backpressure automatically
readable.pipe(writable);

// Transform stream
const { createGzip } = require("zlib");
const gzip = createGzip();
readable.pipe(gzip).pipe(writable); // compress on the fly

// pipeline: better than pipe (handles errors + cleanup)
const { promisify } = require("util");
const pipelineAsync = promisify(pipeline);
await pipelineAsync(createReadStream("input"), createGzip(), createWriteStream("output.gz"));

// Custom transform
const upperCase = new Transform({
  transform(chunk, encoding, callback) {
    callback(null, chunk.toString().toUpperCase());
  }
});
5.4 Express.js (Deep) 
const express = require("express");
const app = express();

// Middleware: (req, res, next) => void
// Runs in order of registration
app.use(express.json());          // parse JSON body
app.use(express.urlencoded({ extended: true })); // parse form data

// Custom middleware
const logger = (req, res, next) => {
  console.log(`${req.method} ${req.url}`);
  next(); // MUST call next() or send response
};
app.use(logger);

// Router organization
const userRouter = express.Router();
userRouter.get("/", listUsers);
userRouter.get("/:id", getUser);
userRouter.post("/", createUser);
userRouter.put("/:id", updateUser);
userRouter.delete("/:id", deleteUser);
app.use("/api/users", userRouter);

// Error-handling middleware: MUST have 4 args
app.use((err, req, res, next) => {
  console.error(err.stack);
  res.status(err.statusCode || 500).json({
    error: err.message || "Internal Server Error"
  });
});

// Async wrapper to catch errors
const asyncHandler = fn => (req, res, next) => Promise.resolve(fn(req, res, next)).catch(next);
app.get("/users/:id", asyncHandler(async (req, res) => {
  const user = await User.findById(req.params.id);
  if (!user) return res.status(404).json({ error: "Not found" });
  res.json(user);
}));

JWT Implementation

import jwt from "jsonwebtoken";

// Sign (create)
const token = jwt.sign(
  { userId: user.id, role: user.role },  // payload
  process.env.JWT_SECRET,                 // secret
  { expiresIn: "15m", algorithm: "HS256" }
);

// Verify
try {
  const decoded = jwt.verify(token, process.env.JWT_SECRET);
  req.user = decoded;
  next();
} catch (err) {
  if (err.name === "TokenExpiredError") return res.status(401).json({ error: "Token expired" });
  return res.status(401).json({ error: "Invalid token" });
}

// Refresh token pattern
// Access token: short-lived (15min), stored in memory
// Refresh token: long-lived (7d), stored in HttpOnly cookie
// Rotation: each refresh issues new refresh token + invalidates old

// JWT vulnerabilities:
// 1. alg: "none" attack → always verify algorithm explicitly
// 2. Weak secrets → use 256-bit random secret
// 3. Sensitive data in payload → payload is base64, NOT encrypted
// 4. Long expiry → use short access tokens + refresh tokens

5.7 Performance & Production 
// PM2: process manager
// pm2 start app.js -i max    -- cluster mode (one process per CPU)
// pm2 reload app             -- zero-downtime reload
// pm2 monit                  -- monitoring dashboard

// Cluster module
const cluster = require("cluster");
const os = require("os");

if (cluster.isPrimary) {
  os.cpus().forEach(() => cluster.fork());
  cluster.on("exit", (worker) => {
    console.log(`Worker ${worker.id} died, forking new`);
    cluster.fork();
  });
} else {
  require("./app"); // each worker runs app
}

⚡ Key Takeaways

  • Node's event loop has 6 phases; setImmediate fires in the check phase, process.nextTick fires between any two phases
  • Streams solve the memory problem of loading entire files/responses into RAM — always prefer streams for large data
  • Never block the event loop with synchronous CPU work; offload to worker threads or child processes
  • cluster spawns multiple processes sharing the same port; each has its own V8 instance and memory space
  • Unhandled promise rejections must be caught globally with process.on('unhandledRejection')
  • Express middleware order matters; error-handling middleware must have exactly 4 parameters: (err, req, res, next)
  • Always use connection pooling for database clients — never create a new connection per request
