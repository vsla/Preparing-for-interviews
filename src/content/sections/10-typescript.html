10. TypeScript (Deep) 2 topics

TypeScript questions appear in most modern frontend and fullstack interviews. Interviewers test generics, conditional types, mapped types, and utility types — not just basic annotations. Understanding structural typing and how TypeScript interacts with JavaScript runtime behavior is the real differentiator.

any vs unknown vs never

let a: any    = "hello"; a.foo();    // no error — opts out of type checking
let u: unknown = "hello"; u.foo();   // Error — must narrow first
if (typeof u === "string") u.toUpperCase(); // OK after narrowing

// never: value that never exists
function fail(msg: string): never { throw new Error(msg); }
function assertExhaustive(x: never): never { throw new Error("Not exhaustive: " + x); }

// Exhaustiveness check with discriminated unions
type Shape = { kind: "circle"; radius: number } | { kind: "square"; side: number };
function area(s: Shape): number {
  switch (s.kind) {
    case "circle": return Math.PI * s.radius ** 2;
    case "square": return s.side ** 2;
    default: return assertExhaustive(s); // Error if new case added
  }
}

Generics

// Basic generic
function identity<T>(arg: T): T { return arg; }

// Constrained generic
function getLength<T extends { length: number }>(arg: T): number {
  return arg.length;
}

// Generic with default
interface Box<T = string> { value: T; }
const b: Box = { value: "hello" }; // T defaults to string

// Generic interface
interface Repository<T, ID = string> {
  findById(id: ID): Promise<T | null>;
  save(entity: T): Promise<T>;
  delete(id: ID): Promise<void>;
}

Utility Types

type User = { id: number; name: string; email: string; role: string; };

type PartialUser   = Partial<User>;                     // all optional
type RequiredUser  = Required<Partial<User>>;           // all required
type ReadonlyUser  = Readonly<User>;                    // all readonly
type UserPreview   = Pick<User, "id" | "name">;         // subset
type UserNoId      = Omit<User, "id">;                  // exclude
type StringOrNum   = Exclude<string | number | boolean, boolean>; // string | number
type OnlyStrNum    = Extract<string | number | boolean, string | number>;
type NotNull       = NonNullable<string | null | undefined>; // string
type Fn = (x: number, y: string) => boolean;
type Ret  = ReturnType<Fn>;       // boolean
type Args = Parameters<Fn>;       // [x: number, y: string]
type PromResult = Awaited<Promise<string>>; // string

Mapped Types

type Optional<T> = { [K in keyof T]?: T[K] };
type Nullable<T> = { [K in keyof T]: T[K] | null };
type Mutable<T>  = { -readonly [K in keyof T]: T[K] }; // remove readonly
type ReadonlyPartial<T> = { readonly [K in keyof T]?: T[K] };

Conditional Types & infer

type IsArray<T> = T extends any[] ? true : false;
type Flatten<T> = T extends Array<infer Item> ? Item : T;
type UnwrapPromise<T> = T extends Promise<infer R> ? R : T;
type FirstArg<T extends (...args: any) => any> =
  T extends (first: infer F, ...rest: any) => any ? F : never;

Template Literal Types & satisfies

type EventName = "click" | "focus" | "blur";
type HandlerName = `on${Capitalize<EventName>}`; // "onClick" | "onFocus" | "onBlur"
type CSSProperties = `${string}-${string}`;

// satisfies: check type without widening
const config = { port: 3000, host: "localhost" } satisfies Record<string, string | number>;
config.port.toFixed(); // TS knows it's number (not widened to string|number)

// as const: infer narrowest type
const routes = ["home", "about", "contact"] as const;
type Route = typeof routes[number]; // "home" | "about" | "contact"

// Module augmentation
declare module "express" {
  interface Request { user?: User; correlationId?: string; }
}
10.2 Tricky TypeScript Questions

Common gotchas

// Q: Why does narrowing not work inside callbacks?
let x: string | null = "hello";
setTimeout(() => {
  x.toUpperCase(); // Error: x might be null (could change before callback)
}, 0);
// Fix: capture value
const copy = x;
setTimeout(() => copy?.toUpperCase(), 0);

// Q: Structural typing — same structure = compatible
interface Cat { meow(): void; }
interface Dog { meow(): void; } // same shape
const cat: Cat = { meow() {} };
const dog: Dog = cat; // OK! TS uses structural (not nominal) typing

// Q: Type widening with const
let x = 3;        // type: number (widened)
const y = 3;      // type: 3 (literal, not widened)
const z = [1,2,3] as const; // type: readonly [1, 2, 3]

// Q: keyof typeof
const obj = { a: 1, b: 2, c: 3 };
type Keys = keyof typeof obj; // "a" | "b" | "c"
function getValue(o: typeof obj, key: Keys) { return o[key]; }

// Q: Conditional distribution
type ToArray<T> = T extends any ? T[] : never;
type Result = ToArray<string | number>; // string[] | number[] (distributes)
// vs:
type NoDistrib<T> = [T] extends [any] ? T[] : never;
type Result2 = NoDistrib<string | number>; // (string | number)[]

⚡ Key Takeaways

  • Structural typing: two types are compatible if their shapes match, regardless of name — duck typing with static analysis
  • Generics + constraints (extends) make code reusable without sacrificing type safety
  • Conditional types (T extends U ? X : Y) and mapped types ({ [K in keyof T]: ... }) enable powerful type transformations
  • unknown is safer than any — you must narrow it before use; any opts out of all type checking
  • as const freezes types to literal values, enabling precise discriminated unions
  • Utility types (Partial, Pick, Omit, ReturnType) eliminate boilerplate type definitions
